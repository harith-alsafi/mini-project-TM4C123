178// Function to read the pressed key's position
179char KeypadWaitForChar() {
180 while (1) {
181 for (int col = 0; col < 4; col++) {
182 // Set the current column to HIGH
183 KEYPAD_COL_PIN = (1 << col);
184 DelayMicro(2);
185
186 // Check each row
187 for (int row = 0; row < 4; row++) {
188 if (KEYPAD_ROW_PIN & (1 << row)) {
189 return KEYPAD_KEYS[row][col];
190 }
191 }
192
193 KEYPAD_COL_PIN = 0x00;
194 }
195 }
196}
