167// Function to read the pressed key's position
168char KeypadGetChar() {
169 while (1) {
170 for (int col = 0; col < 4; col++) {
171 // Set the current column to HIGH
172 KEYPAD_COL_PIN = (1 << col);
173 DelayMicro(2);
174
175 // Check each row
176 for (int row = 0; row < 4; row++) {
177 if (KEYPAD_ROW_PIN & (1 << row)) {
178 return KEYPAD_KEYS[row][col];
179 }
180 }
181
182 KEYPAD_COL_PIN = 0x00;
183 }
184 }
185}
