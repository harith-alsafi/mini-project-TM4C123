407 // display normally
408 else if(!isShiftOn && key.normalDisplayType == DISPLAY_NORMAL || isShiftOn && key.shiftDisplayType == DISPLAY_NORMAL){
409 RunUserInput(key);
410 continue;
411 }
412 // clear input
413 else if(function == CLEAR_ALL){
414 ClearAllInput(true);
415 continue;
416 }
417 // clear last input
418 else if(function == CLEAR_LAST){
419 ClearLastInput();
420 continue;
421 }
422 // // end input
423 else if(function == END_INPUT){
424 // calculate input
425 if(mode == CALCULATOR_MODE){
426 RunCalculator();
427 continue;
428 }
429 // try to login / confirm change password
430 else if(mode == LOGIN_MODE){
431 RunLogin();
432 continue;
433 }
434 // change the password and go back
435 else if(mode == CHANGE_PASSWORD) {
436 RunChangePassword();
437 continue;
438 }
439 }
440 // go back to calculator mode
441 else if(function == GO_BACK && mode == CHANGE_PASSWORD){
442 mode = CALCULATOR_MODE;
443 isShiftOn = false;
444 RefreshDisplay();
445 }
446 // open change password
447 else if(function == OPEN_CHANGE_PASSWORD && mode == CALCULATOR_MODE){
448 mode = CHANGE_PASSWORD;
449 isShiftOn = false;
450 RefreshDisplay();
451 }
452 }
